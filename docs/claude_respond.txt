function getBrowserHeaders() {
  const pick = UA_POOL[Math.floor(Math.random() * UA_POOL.length)]
  return {
    'User-Agent':         pick.ua,
    'Accept-Language':    'en-US,en;q=0.9',
    'Accept-Encoding':    'gzip, deflate, br, zstd',
    'Sec-Ch-Ua':          pick.ch,
    'Sec-Ch-Ua-Mobile':   '?0',
    'Sec-Ch-Ua-Platform': pick.platform,
  }
}
